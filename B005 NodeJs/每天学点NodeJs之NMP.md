# 常用命令

```shell
# 安装
npm install <name> [-g] # -g 全局安装
# 卸载
npm uninstall [-g] <name>
# 更新
npm update <name> [-g]
# 更新所有
npm update

# 帮助命令
npm help

npm list <name> [-g] [--depth n]
npm ls [-g]


# 全局模块的存放路径
npm config get prefix
npm config set prefix "{path}\nodejs\node_global"
# cache 的路径
npm config get cache
npm config set cache "{path}\nodejs\node_cache"



```
